void ySort(float d[],char s,char e){
	char i=s+1,j;
	float k;
	while(i<=e){
		k=d[i];
		j=i-1;
		i++;
		while(j>=s && d[j]>k){
			d[j+1]=d[j];
			j--;
		}
		d[j+1]=k;
	}
}
