a = np.array([[123,456,678],[3214,456,134]])
b = np.array([[1235,3124,432],[43,13,134]])
拼接合并
列对齐（默认）	
c = numpy.concatenate((a,b),axis = 0)
array([[ 123,  456,  678],
       [3214,  456,  134],
       [1235, 3124,  432],
       [  43,   13,  134]])
行对齐
c = numpy.concatenate((a,b),axis = 1)
array([[ 123,  456,  678, 1235, 3124,  432],
       [3214,  456,  134,   43,   13,  134]])
